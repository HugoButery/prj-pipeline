pipeline {
   agent any
    environment {
        GitBranch = "main"
        StateBucketName = "dev-ops-grp4"
    }
    
   stages {
        stage('Clean Jenkins Workspace') {
            steps {
                cleanWs()
            }
        }
        
        stage('check Vars') {
            steps {
                sh "echo check first if vars are compliant"
                    
            }
        }
        
        stage('checkout git') {
            steps {
                sh "pwd"
                dir("Deploy_WebApp") {
                    sh "pwd"
                    git(
                        url: 'https://github.com/HugoButery/CICD_Projet_DevOps',
                        credentialsId: 'gh-cred',
                        branch: "${GitBranch}"
                    )
                    sh "ls"
                }
            }
        }
        
        stage('Deploy WebApp') {
            steps {
                dir("Deploy_WebApp") {
                dir("Terraform"){
                dir("App"){
                    sh "terraform --version"
                    sh "pwd"
                    sh "ls"

                    wrap([$class: 'AnsiColorBuildWrapper', 'colorMapName': 'xterm']) {
                        sh "terraform init -backend-config='bucket=$StateBucketName' -backend-config='key=$APP_NAME/$ENV/app.tfstate' -backend-config='region=eu-west-1' -var 'env=$ENV' -var 'app_name=$APP_NAME'"
                        sh "terraform apply -auto-approve"
                    }
                }
                }
                }
            }
        }
    }
}
