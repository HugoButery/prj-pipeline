pipeline {
   agent any
    environment {
        GitBranch = "main"
    }
    
   stages {
        stage('Clean Jenkins Workspace') {
            steps {
                cleanWs()
            }
        }
        
        stage('check Vars') {
            steps {
                sh "echo check first if vars are compliant"
                    
            }
        }
        
        stage('checkout git') {
            steps {
                sh "pwd"
                dir("Deploy_Infra") {
                    sh "pwd"
                    git(
                        url: 'https://github.com/HugoButery/CICD_Projet_DevOps',
                        credentialsId: 'gh-cred',
                        branch: "${GitBranch}"
                    )
                    sh "ls"
                }
            }
        }
        
        stage('Destroy Infra') {
            steps {
                dir("Deploy_Infra") {
                dir("Terraform"){
                    sh "pwd"
                    sh "ls"
                    
                    wrap([$class: 'AnsiColorBuildWrapper', 'colorMapName': 'xterm']) {
                        sh "terraform init -backend-config='dev-ops-grp4' -backend-config='key=architecture-grp4' -backend-config='region=eu-west-1'"
                        sh "terraform destroy -auto-approve"
                    }
                    }
                }
            }
        }
    }
}
