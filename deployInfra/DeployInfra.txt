pipeline {
   agent any
    environment {
        GitBranch = "main"
    }
    

    
   stages {
        stage('Clean Jenkins Workspace before start') {
            steps {
                cleanWs()
            }
        }
        
        stage('check Vars') {
            steps {
                sh "echo $APP_NAME"
                sh "echo $ENV"
            }
        }
        
        stage('checkout git') {
            steps {
                sh "pwd"
                dir("Deploy_Infra") {
                    sh "pwd"
                    git(
                        url: 'https://github.com/HugoButery/CICD_Projet_DevOps',
                        credentialsId: 'gh-cred',
                        branch: "${GitBranch}"
                    )
                    sh "ls"
                }
            }
        }
        
        stage('Deploy Infra') {
            steps {
               dir("Deploy_Infra"){
                dir("Terraform") {
                dir("Infra"){
                    sh "pwd"
                    
                    sh "ls"

                    wrap([$class: 'AnsiColorBuildWrapper', 'colorMapName': 'xterm']) {
                        sh "terraform init -backend-config='bucket=dev-ops-grp4' -backend-config='key=$APP_NAME/$ENV/infra.tfstate' -backend-config='region=eu-west-1' -var 'env=$ENV' -var 'app_name=$APP_NAME'"
                        sh "terraform apply -auto-approve"
                    }
                }
                }
               }
            }
        }
    }
}
